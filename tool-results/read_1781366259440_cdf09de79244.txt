    61→function runCommand(command, args = [], options = {}) {
    62→  return new Promise((resolve) => {
    63→    const child = spawn(command, args, { shell: false, windowsHide: true, ...options });
    64→    let stdout = '';
    65→    let stderr = '';
    66→    child.stdout?.on('data', data => { stdout += data.toString(); });
    67→    child.stderr?.on('data', data => { stderr += data.toString(); });
    68→    child.on('error', err => resolve({ success: false, error: err.message, stdout, stderr }));
    69→    child.on('close', code => resolve({ success: code === 0, code, stdout, stderr, error: code === 0 ? '' : (stderr || `Exit code ${code}`) }));
    70→  });
    71→}

    97→async function executeDesktopAction(action = {}) {
    98→  const type = action.type || action.action;
    99→  if (type === 'open-url' || type === 'open_url') {
   100→    await shell.openExternal(action.url);
   101→    return { success: true, message: 'Opened URL' };
   102→  }
   103→  if (type === 'search-google' || type === 'google_search') {
   104→    await shell.openExternal('https://www.google.com/search?q=' + encodeURIComponent(action.query || ''));
   105→    return { success: true, message: 'Google search opened' };
   106→  }
   107→  if (type === 'open-youtube' || type === 'play-youtube' || type === 'search-youtube' || type === 'youtube_search') {
   108→    const query = action.query || '';
   109→    const url = query
   110→      ? 'https://www.youtube.com/results?search_query=' + encodeURIComponent(query)
   111→      : 'https://www.youtube.com';
   112→    await shell.openExternal(url);
   113→    return { success: true, message: query ? 'YouTube search opened' : 'YouTube opened' };
   114→  }
   115→  if (type === 'open-app' || type === 'open_app') {
   116→    exec(process.platform === 'win32' ? `start "" "${normalizeAppName(action.app || action.name)}"` : `open -a "${action.app || action.name}"`);
   117→    return { success: true, message: `Opened ${action.app || action.name}` };
   118→  }
   119→  if (type === 'play-audio' || type === 'play_audio') {
   120→    // Open audio URL in default browser/player
   121→    const audioUrl = action.url || '';
   122→    if (audioUrl) {
   123→      await shell.openExternal(audioUrl);
   124→      return { success: true, message: 'Audio opened' };
   125→    }
   126→    // If no URL but has query, search on YouTube
   127→    if (action.query) {
   128→      const url = 'https://www.youtube.com/results?search_query=' + encodeURIComponent(action.query);
   129→      await shell.openExternal(url);
   130→      return { success: true, message: 'Audio search opened on YouTube' };
   131→    }
   132→    return { success: false, message: 'No audio URL or query provided' };
   133→  }
   134→  if (type === 'open-whatsapp' || type === 'whatsapp') {
   135→    const phone = action.phone || action.number || '';
   136→    const message = action.message || action.text || '';
   137→    const url = phone
   138→      ? `https://web.whatsapp.com/send?phone=${encodeURIComponent(phone)}${message ? '&text=' + encodeURIComponent(message) : ''}`
   139→      : 'https://web.whatsapp.com';
   140→    await shell.openExternal(url);
   141→    return { success: true, message: phone ? 'WhatsApp chat opened' : 'WhatsApp Web opened' };
   142→  }
   143→  if (type === 'volume-up' || type === 'volume_up') {
   144→    if (process.platform === 'win32') return { ...(await sendVolumeKey(175, action.steps || 5)), message: 'Volume increased' };
   145→  }
   146→  if (type === 'volume-down' || type === 'volume_down') {
   147→    if (process.platform === 'win32') return { ...(await sendVolumeKey(174, action.steps || 5)), message: 'Volume decreased' };
   148→  }
   149→  if (type === 'mute-toggle' || type === 'mute_toggle' || type === 'mute' || type === 'unmute') {
   150→    if (process.platform === 'win32') return { ...(await sendVolumeKey(173, 1)), message: 'Mute toggled' };
   151→  }
   152→  if (type === 'screenshot') {
   153→    const file = action.path || path.join(app.getPath('pictures'), `jarvis-screenshot-${Date.now()}.png`);
   154→    const escaped = file.replace(/'/g, "''");
   155→    const result = await ps(`Add-Type -AssemblyName System.Windows.Forms; Add-Type -AssemblyName System.Drawing; $b=[System.Windows.Forms.Screen]::PrimaryScreen.Bounds; $bmp=New-Object System.Drawing.Bitmap $b.Width,$b.Height; $g=[System.Drawing.Graphics]::FromImage($bmp); $g.CopyFromScreen($b.Location,[System.Drawing.Point]::Empty,$b.Size); $bmp.Save('${escaped}', [System.Drawing.Imaging.ImageFormat]::Png); $g.Dispose(); $bmp.Dispose(); Write-Output '${escaped}'`);
   156→    return { success: result.success, message: `Screenshot saved: ${file}`, path: file, error: result.error };
   157→  }
   158→  if (type === 'system-command') {
   159→    const allowed = new Set(['lock']);
   160→    if (!allowed.has(action.command)) return { success: false, message: 'Command not allowed from UI' };
   161→    if (process.platform === 'win32') exec('rundll32.exe user32.dll,LockWorkStation');
   162→    return { success: true, message: 'Command sent' };
   163→  }
   164→  return { success: false, message: 'Unknown desktop action' };
   165→}

   167→async function listenWithWindowsSpeech(language = 'en-US') {
   168→  if (process.platform !== 'win32') {
   169→    return { success: false, error: 'Windows Speech API is only available on Windows' };
   170→  }
   171→  const culture = String(language).startsWith('ur') ? 'ur-PK' : 'en-US';
   172→  const script = `
   173→Add-Type -AssemblyName System.Speech
   174→$culture = New-Object System.Globalization.CultureInfo('${culture}')
   175→$recognizer = New-Object System.Speech.Recognition.SpeechRecognitionEngine($culture)
   176→$recognizer.SetInputToDefaultAudioDevice()
   177→$dictation = New-Object System.Speech.Recognition.DictationGrammar
   178→$recognizer.LoadGrammar($dictation)
   179→$result = $recognizer.Recognize([TimeSpan]::FromSeconds(8))
   180→if ($result -and $result.Text) { Write-Output $result.Text }
   181→$recognizer.Dispose()
   182→`;
   183→  const result = await ps(script);
   184→  const text = (result.stdout || '').trim();
   185→  return text ? { success: true, text, method: 'windows-speech' } : { success: false, error: result.error || 'No speech recognized' };
   186→}

   228→function attachProcessLogging(child, name, onStdout, onStderr) {
   229→  const attach = (stream, level, handler) => {
   230→    if (!stream || stream.destroyed || stream.readableEnded) return;
   231→    stream.on('data', (data) => {
   232→      try {
   233→        handler(data.toString());
   234→      } catch (err) {
   235→        if (!isPipeClosedError(err)) log('WARN', `[${name}] Stream handler failed:`, err.message);
   236→      }
   237→    });
   238→    stream.on('error', (err) => {
   239→      if (!isPipeClosedError(err)) log(level, `[${name}] Stream error:`, err.message);
   240→    });
   241→  };
   242→  attach(child?.stdout, 'WARN', onStdout);
   243→  attach(child?.stderr, 'WARN', onStderr);
   244→}
   245→log('INFO', '═══════════════════════════════════════');

   274→async function launchCloudBackend() {
   275→  return new Promise(async (resolve, reject) => {
   276→    log('INFO', '[Launcher] Starting cloud backend...');
   277→
   278→    const root = getProjectRoot();
   279→    const isWindows = process.platform === 'win32';
   280→
   281→    // Check if npm is available
   282→    const npmCmd = isWindows ? 'npm.cmd' : 'npm';
   283→
   284→    try {
   285→      if (await checkCloudHealth()) {
   286→        cloudReady = true;
   287→        cloudError = '';
   288→        log('INFO', '[Launcher] Existing cloud backend detected');
   289→        resolve(true);
   290→        return;
   291→      }
   292→
   293→      cloudProcess = spawn(npmCmd, ['run', 'dev', '--', '--hostname', '127.0.0.1', '--port', '3000'], {
   294→        cwd: root,
   295→        shell: true,
   296→        stdio: ['ignore', 'pipe', 'pipe'],
   297→      });
   298→
   299→      attachProcessLogging(cloudProcess, 'Cloud', (output) => {
   300→        log('INFO', '[Cloud]', output.trim());
   301→
   302→        // Check for ready signal
   303→        if (output.includes('ready') || output.includes('started server') || output.includes('Local:')) {
   304→          waitForCloud().then((ready) => {
   305→            cloudReady = ready;
   306→            cloudError = ready ? '' : 'Next.js started, but /api/health did not answer on http://127.0.0.1:3000';
   307→            log(ready ? 'INFO' : 'WARN', '[Launcher]', ready ? 'Cloud backend ready' : cloudError);
   308→            resolve(ready);
   309→          });
   310→        }
   311→      }, (output) => {
   312→        output = output.trim();
   313→        if (output) cloudError = output.split(/\r?\n/).slice(-1)[0];
   314→        log('WARN', '[Cloud]', output);
   315→      });
   316→
   317→      cloudProcess.on('error', (err) => {
   318→        cloudError = err.message;
   319→        log('ERROR', '[Cloud] Process error:', err.message);
   320→        reject(err);
   321→      });
   322→
   323→      cloudProcess.on('close', (code) => {
   324→        log('INFO', '[Cloud] Process exited with code:', code);
   325→        cloudReady = false;
   326→        if (code !== 0 && code !== null) cloudError = `Next.js process exited with code ${code}`;
   327→      });
   328→
   329→      // Timeout fallback
   330→      setTimeout(() => {
   331→        if (!cloudReady) {
   332→          cloudError = cloudError || 'Next.js startup timed out on http://127.0.0.1:3000';
   333→          log('WARN', '[Launcher]', cloudError);
   334→          resolve(false);
   335→        }
   336→      }, MAX_STARTUP_WAIT);
   337→
   338→    } catch (err) {
   339→      cloudError = err.message;
   340→      log('ERROR', '[Launcher] Failed to start cloud:', err.message);
   341→      reject(err);
   342→    }
   343→  });
   344→}

   797→// ─── Voice IPC: Direct STT & TTS from Main Process ───
   798→// These bypass the renderer's fetch() calls which can fail in Electron
   799→
   800→function buildMultipart(fields, boundary) {
   801→  const parts = [];
   802→  for (const [name, value] of Object.entries(fields)) {
   803→    if (value && typeof value === 'object' && value.filename) {
   804→      parts.push(Buffer.from(`--${boundary}\r\nContent-Disposition: form-data; name="${name}"; filename="${value.filename}"\r\nContent-Type: ${value.contentType || 'application/octet-stream'}\r\n\r\n`, 'utf8'));
   805→      parts.push(Buffer.isBuffer(value.data) ? value.data : Buffer.from(value.data));
   806→      parts.push(Buffer.from('\r\n', 'utf8'));
   807→    } else {
   808→      parts.push(Buffer.from(`--${boundary}\r\nContent-Disposition: form-data; name="${name}"\r\n\r\n${value}\r\n`, 'utf8'));
   809→    }
   810→  }
   811→  parts.push(Buffer.from(`--${boundary}--\r\n`, 'utf8'));
   812→  return Buffer.concat(parts);
   813→}

   815→function httpsPostBuffer(urlStr, body, headers) {
   816→  return new Promise((resolve, reject) => {
   817→    const parsedUrl = new URL(urlStr);
   818→    const mod = parsedUrl.protocol === 'http:' ? require('http') : require('https');
   819→    const options = {
   820→      hostname: parsedUrl.hostname,
   821→      port: parsedUrl.port || (parsedUrl.protocol === 'http:' ? 80 : 443),
   822→      path: parsedUrl.pathname + parsedUrl.search,
   823→      method: 'POST',
   824→      headers: { ...headers, 'Content-Length': body.length },
   825→    };
   826→    const req = mod.request(options, (res) => {
   827→      const chunks = [];
   828→      res.on('data', (chunk) => chunks.push(chunk));
   829→      res.on('end', () => {
   830→        const data = Buffer.concat(chunks);
   831→        resolve({ statusCode: res.statusCode, headers: res.headers, data });
   832→      });
   833→    });
   834→    req.on('error', reject);
   835→    req.setTimeout(30000, () => { req.destroy(new Error('Request timeout')); });
   836→    req.write(body);
   837→    req.end();
   838→  });
   839→}

   841→// IPC: Transcribe audio (base64) via Groq/OpenAI Whisper
   842→ipcMain.handle('transcribe-audio-base64', async (_, base64Audio, language = 'ur', apiKeys = {}) => {
   843→  try {
   844→    const audioBuffer = Buffer.from(base64Audio, 'base64');
   845→    const groqKey = apiKeys.groq || process.env.GROQ_API_KEY || '';
   846→    const openaiKey = apiKeys.openai || process.env.OPENAI_API_KEY || '';
   847→
   848→    if (!groqKey && !openaiKey) {
   849→      return { success: false, error: 'Groq یا OpenAI API Key ضروری ہے۔ Settings میں جا کر API Key ڈالیں۔' };
   850→    }
   851→
   852→    if (audioBuffer.length < 100) {
   853→      return { success: false, error: 'آڈیو بہت چھوٹا ہے۔ دوبارہ مائیک پر بولیں۔' };
   854→    }
   855→
   856→    log('INFO', '[STT-IPC] Starting transcription, audio size:', audioBuffer.length, 'language:', language);
   857→
   858→    // Detect audio format from binary header
   859→    let filename = 'voice.webm';
   860→    let contentType = 'audio/webm';
   861→    if (audioBuffer.length > 4) {
   862→      const header = audioBuffer.slice(0, 12).toString('hex');
   863→      if (header.startsWith('52494646')) { // RIFF = WAV
   864→        filename = 'voice.wav';
   865→        contentType = 'audio/wav';
   866→        log('INFO', '[STT-IPC] Detected WAV format');
   867→      } else if (header.startsWith('fff3') || header.startsWith('fffb') || header.startsWith('ffe3') || header.startsWith('49443303')) { // MP3 / ID3
   868→        filename = 'voice.mp3';
   869→        contentType = 'audio/mpeg';
   870→        log('INFO', '[STT-IPC] Detected MP3 format');
   871→      } else if (header.startsWith('1a45dfa3') || header.includes('a1')) { // WebM / MKV
   872→        filename = 'voice.webm';
   873→        contentType = 'audio/webm';
   874→        log('INFO', '[STT-IPC] Detected WebM/MKV format');
   875→      } else if (header.startsWith('4f676753')) { // OGG
   876→        filename = 'voice.ogg';
   877→        contentType = 'audio/ogg';
   878→        log('INFO', '[STT-IPC] Detected OGG format');
   879→      } else {
   880→        log('INFO', '[STT-IPC] Unknown audio format, defaulting to webm. Header:', header.substring(0, 16));
   881→      }
   882→    }
   883→
   884→    // Map language codes to Whisper-compatible codes
   885→    const whisperLang = language === 'ur' ? 'ur' : language === 'en' ? 'en' : language || 'ur';
   886→
   887→    // Try Groq Whisper first (fastest, cheapest, best for Urdu)
   888→    if (groqKey) {
   889→      try {
   890→        const boundary = '----JarvisSTT' + Date.now();
   891→        const body = buildMultipart({
   892→          file: { data: audioBuffer, filename, contentType },
   893→          model: 'whisper-large-v3-turbo',
   894→          language: whisperLang,
   895→          response_format: 'json',
   896→        }, boundary);
   897→
   898→        const result = await httpsPostBuffer('https://api.groq.com/openai/v1/audio/transcriptions', body, {
   899→          'Authorization': `Bearer ${groqKey}`,
   900→          'Content-Type': `multipart/form-data; boundary=${boundary}`,
   901→        });
   902→
   903→        log('INFO', '[STT-IPC] Groq response:', result.statusCode, 'size:', result.data.length);
   904→
   905→        if (result.statusCode === 200) {
   906→          try {
   907→            const json = JSON.parse(result.data.toString('utf8'));
   908→            if (json.text && json.text.trim()) {
   909→              log('INFO', '[STT-IPC] Groq Whisper success:', json.text.substring(0, 80));
   910→              return { success: true, text: json.text.trim(), method: 'groq-whisper-ipc', language: json.language || whisperLang };
   911→            }
   912→            log('WARN', '[STT-IPC] Groq returned empty text');
   913→          } catch (parseErr) {
   914→            log('WARN', '[STT-IPC] Groq JSON parse error:', parseErr.message);
   915→          }
   916→        } else {
   917→          const errBody = result.data.toString('utf8').substring(0, 300);
   918→          log('WARN', '[STT-IPC] Groq failed:', result.statusCode, errBody);
   919→          // If 401, API key is invalid — don't waste time retrying
   920→          if (result.statusCode === 401) {
   921→            return { success: false, error: 'Groq API Key غلط ہے۔ Settings میں درست Key ڈالیں۔' };
   922→          }
   923→        }
   924→      } catch (err) {
   925→        log('WARN', '[STT-IPC] Groq error:', err.message);
   926→      }
   927→    }
   928→
   929→    // Try OpenAI Whisper (better quality, more expensive)
   930→    if (openaiKey) {
   931→      try {
   932→        const boundary = '----JarvisSTT' + Date.now();
   933→        const body = buildMultipart({
   934→          file: { data: audioBuffer, filename, contentType },
   935→          model: 'whisper-1',

  1027→function httpsGetBuffer(urlStr, headers) {
  1028→  return new Promise((resolve, reject) => {
  1029→    const parsedUrl = new URL(urlStr);
  1030→    const mod = parsedUrl.protocol === 'http:' ? require('http') : require('https');
  1031→    const options = {
  1032→      hostname: parsedUrl.hostname,
  1033→      port: parsedUrl.port || (parsedUrl.protocol === 'http:' ? 80 : 443),
  1034→      path: parsedUrl.pathname + parsedUrl.search,
  1035→      method: 'GET',
  1036→      headers: headers || {},
  1037→    };
  1038→    const req = mod.request(options, (res) => {
  1039→      const chunks = [];
  1040→      res.on('data', (chunk) => chunks.push(chunk));
  1041→      res.on('end', () => {
  1042→        const data = Buffer.concat(chunks);
  1043→        resolve({ statusCode: res.statusCode, headers: res.headers, data });
  1044→      });
  1045→    });
  1046→    req.on('error', reject);
  1047→    req.setTimeout(15000, () => { req.destroy(new Error('Request timeout')); });
  1048→    req.end();
  1049→  });
  1050→}

  1109→async function ensurePiperBinary() {
  1110→  try {
  1111→    const piperBin = process.platform === 'win32' ? 'piper.exe' : 'piper';
  1112→    const localPiperPath = path.join(PIPER_BIN_DIR, piperBin);
  1113→
  1114→    // Already downloaded?
  1115→    if (fs.existsSync(localPiperPath)) {
  1116→      return { ready: true, path: localPiperPath };
  1117→    }
  1118→
  1119→    // Check PATH
  1120→    try {
  1121→      const whichResult = await runCommand(process.platform === 'win32' ? 'where' : 'which', [piperBin]);
  1122→      if (whichResult.success && whichResult.stdout.trim()) {
  1123→        const foundPath = whichResult.stdout.trim().split('\n')[0].trim();
  1124→        return { ready: true, path: foundPath };
  1125→      }
  1126→    } catch {}
  1127→
  1128→    // Auto-download Piper binary for Windows
  1129→    if (process.platform === 'win32') {
  1130→      log('INFO', '[Piper] Auto-downloading piper.exe...');
  1131→      if (!fs.existsSync(PIPER_BIN_DIR)) {
  1132→        fs.mkdirSync(PIPER_BIN_DIR, { recursive: true });
  1133→      }
  1134→
  1135→      // Download Piper Windows release
  1136→      const piperVersion = '2023.11.14-2';
  1137→      const piperUrl = `https://github.com/rhasspy/piper/releases/download/${piperVersion}/piper_windows_amd64.zip`;
  1138→      const zipPath = path.join(PIPER_BIN_DIR, 'piper.zip');
  1139→
  1140→      const downloadResult = await downloadFile(piperUrl, zipPath);
  1141→      if (!downloadResult.success) {
  1142→        log('WARN', '[Piper] Download failed:', downloadResult.error);
  1143→        return { ready: false, error: 'Piper binary download failed: ' + downloadResult.error };
  1144→      }
  1145→
  1146→      // Extract using PowerShell
  1147→      const extractResult = await ps(`Expand-Archive -Path '${zipPath}' -DestinationPath '${PIPER_BIN_DIR}' -Force`);
  1148→      try { fs.unlinkSync(zipPath); } catch {}
  1149→
  1150→      if (extractResult.success || fs.existsSync(localPiperPath)) {
  1151→        log('INFO', '[Piper] Binary downloaded and extracted successfully');
  1152→        return { ready: true, path: localPiperPath };
  1153→      }
  1154→
  1155→      // Check if piper.exe is in a subdirectory
  1156→      const subDir = path.join(PIPER_BIN_DIR, 'piper');
  1157→      const subPiperPath = path.join(subDir, piperBin);
  1158→      if (fs.existsSync(subPiperPath)) {
  1159→        // Move to expected location
  1160→        try {
  1161→          const files = fs.readdirSync(subDir);
  1162→          for (const file of files) {
  1163→            const src = path.join(subDir, file);
  1164→            const dest = path.join(PIPER_BIN_DIR, file);
  1165→            if (!fs.existsSync(dest)) {
  1166→              fs.copyFileSync(src, dest);
  1167→            }
  1168→          }
  1169→        } catch {}
  1170→        if (fs.existsSync(localPiperPath)) {
  1171→          return { ready: true, path: localPiperPath };
  1172→        }
  1173→        return { ready: true, path: subPiperPath };
  1174→      }
  1175→
  1176→      return { ready: false, error: 'Piper binary not found after extraction' };
  1177→    }
  1178→
  1179→    return { ready: false, error: 'Piper binary auto-download only supported on Windows. Install manually.' };
  1180→  } catch (err) {
  1181→    log('WARN', '[Piper] Binary check error:', err.message);
  1182→    return { ready: false, error: err.message };
  1183→  }

  1737→function downloadFile(urlStr, destPath) {
  1738→  return new Promise((resolve) => {
  1739→    try {
  1740→      const parsedUrl = new URL(urlStr);
  1741→      const mod = parsedUrl.protocol === 'http:' ? require('http') : require('https');
  1742→
  1743→      const followRedirect = (url, redirectCount = 0) => {
  1744→        if (redirectCount > 5) {
  1745→          resolve({ success: false, error: 'Too many redirects' });
  1746→          return;
  1747→        }
  1748→        const pu = new URL(url);
  1749→        const m = pu.protocol === 'http:' ? require('http') : require('https');
  1750→        const opts = {
  1751→          hostname: pu.hostname,
  1752→          port: pu.port || (pu.protocol === 'http:' ? 80 : 443),
  1753→          path: pu.pathname + pu.search,
  1754→          method: 'GET',
  1755→          headers: { 'User-Agent': 'JARVIS-Hybrid-Desktop/3.0.3' },
  1756→        };
  1757→        const req = m.request(opts, (res) => {
  1758→          if (res.statusCode >= 300 && res.statusCode < 400 && res.headers.location) {
  1759→            followRedirect(res.headers.location, redirectCount + 1);
  1760→            return;
  1761→          }
  1762→          if (res.statusCode !== 200) {
  1763→            resolve({ success: false, error: `HTTP ${res.statusCode}` });
  1764→            return;
  1765→          }
  1766→          const file = fs.createWriteStream(destPath);
  1767→          res.pipe(file);
  1768→          file.on('finish', () => {
  1769→            file.close();
  1770→            resolve({ success: true });
  1771→          });
  1772→          file.on('error', (err) => {
  1773→            try { fs.unlinkSync(destPath); } catch {}
  1774→            resolve({ success: false, error: err.message });
  1775→          });
  1776→        });
  1777→        req.on('error', (err) => resolve({ success: false, error: err.message }));
  1778→        req.setTimeout(300000, () => { req.destroy(new Error('Download timeout')); resolve({ success: false, error: 'Download timeout' }); });
  1779→        req.end();
  1780→      };
  1781→
  1782→      followRedirect(urlStr);
  1783→    } catch (err) {
  1784→      resolve({ success: false, error: err.message });
  1785→    }
  1786→  });
  1787→}
