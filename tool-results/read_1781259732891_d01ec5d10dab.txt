    61→    61→function runCommand(command, args = [], options = {}) {
    62→    62→  return new Promise((resolve) => {
    63→    63→    const child = spawn(command, args, { shell: false, windowsHide: true, ...options });
    64→    64→    let stdout = '';
    65→    65→    let stderr = '';
    66→    66→    child.stdout?.on('data', data => { stdout += data.toString(); });
    67→    67→    child.stderr?.on('data', data => { stderr += data.toString(); });
    68→    68→    child.on('error', err => resolve({ success: false, error: err.message, stdout, stderr }));
    69→    69→    child.on('close', code => resolve({ success: code === 0, code, stdout, stderr, error: code === 0 ? '' : (stderr || `Exit code ${code}`) }));
    70→    70→  });
    71→    71→}

    97→    97→async function executeDesktopAction(action = {}) {

    99→    99→  if (type === 'open-url' || type === 'open_url') {
   100→   100→    await shell.openExternal(action.url);

   102→   102→  }
   103→   103→  if (type === 'search-google' || type === 'google_search') {
   104→   104→    await shell.openExternal('https://www.google.com/search?q=' + encodeURIComponent(action.query || ''));

   107→   107→  if (type === 'open-youtube' || type === 'play-youtube' || type === 'search-youtube' || type === 'youtube_search') {
   108→   108→    const query = action.query || '';
   109→   109→    const url = query
   110→   110→      ? 'https://www.youtube.com/results?search_query=' + encodeURIComponent(query)
   111→   111→      : 'https://www.youtube.com';

   115→   115→  if (type === 'open-app' || type === 'open_app') {

   118→   118→  }
   119→   119→  if (type === 'play-audio' || type === 'play_audio') {
   120→   120→    // Open audio URL in default browser/player
   121→   121→    const audioUrl = action.url || '';
   122→   122→    if (audioUrl) {
   123→   123→      await shell.openExternal(audioUrl);
   124→   124→      return { success: true, message: 'Audio opened' };
   125→   125→    }
   126→   126→    // If no URL but has query, search on YouTube
   127→   127→    if (action.query) {
   128→   128→      const url = 'https://www.youtube.com/results?search_query=' + encodeURIComponent(action.query);
   129→   129→      await shell.openExternal(url);
   130→   130→      return { success: true, message: 'Audio search opened on YouTube' };

   132→   132→    return { success: false, message: 'No audio URL or query provided' };
   133→   133→  }
   134→   134→  if (type === 'open-whatsapp' || type === 'whatsapp') {
   135→   135→    const phone = action.phone || action.number || '';
   136→   136→    const message = action.message || action.text || '';
   137→   137→    const url = phone
   138→   138→      ? `https://web.whatsapp.com/send?phone=${encodeURIComponent(phone)}${message ? '&text=' + encodeURIComponent(message) : ''}`
   139→   139→      : 'https://web.whatsapp.com';
   140→   140→    await shell.openExternal(url);
   141→   141→    return { success: true, message: phone ? 'WhatsApp chat opened' : 'WhatsApp Web opened' };
   142→   142→  }
   143→   143→  if (type === 'volume-up' || type === 'volume_up') {
   144→   144→    if (process.platform === 'win32') return { ...(await sendVolumeKey(175, action.steps || 5)), message: 'Volume increased' };
   145→   145→  }
   146→   146→  if (type === 'volume-down' || type === 'volume_down') {
   147→   147→    if (process.platform === 'win32') return { ...(await sendVolumeKey(174, action.steps || 5)), message: 'Volume decreased' };
   148→   148→  }
   149→   149→  if (type === 'mute-toggle' || type === 'mute_toggle' || type === 'mute' || type === 'unmute') {
   150→   150→    if (process.platform === 'win32') return { ...(await sendVolumeKey(173, 1)), message: 'Mute toggled' };
   151→   151→  }
   152→   152→  if (type === 'screenshot') {
   153→   153→    const file = action.path || path.join(app.getPath('pictures'), `jarvis-screenshot-${Date.now()}.png`);
   154→   154→    const escaped = file.replace(/'/g, "''");

   156→   156→    return { success: result.success, message: `Screenshot saved: ${file}`, path: file, error: result.error };

   159→   159→    const allowed = new Set(['lock']);
   160→   160→    if (!allowed.has(action.command)) return { success: false, message: 'Command not allowed from UI' };
   161→   161→    if (process.platform === 'win32') exec('rundll32.exe user32.dll,LockWorkStation');
   162→   162→    return { success: true, message: 'Command sent' };

   165→   165→}

   228→   228→function attachProcessLogging(child, name, onStdout, onStderr) {
   229→   229→  const attach = (stream, level, handler) => {
   230→   230→    if (!stream || stream.destroyed || stream.readableEnded) return;
   231→   231→    stream.on('data', (data) => {
   232→   232→      try {
   233→   233→        handler(data.toString());
   234→   234→      } catch (err) {
   235→   235→        if (!isPipeClosedError(err)) log('WARN', `[${name}] Stream handler failed:`, err.message);
   236→   236→      }
   237→   237→    });
   238→   238→    stream.on('error', (err) => {
   239→   239→      if (!isPipeClosedError(err)) log(level, `[${name}] Stream error:`, err.message);
   240→   240→    });
   241→   241→  };
   242→   242→  attach(child?.stdout, 'WARN', onStdout);
   243→   243→  attach(child?.stderr, 'WARN', onStderr);
   244→   244→}

   274→   274→async function launchCloudBackend() {
   275→   275→  return new Promise(async (resolve, reject) => {
   276→   276→    log('INFO', '[Launcher] Starting cloud backend...');
   277→   277→
   278→   278→    const root = getProjectRoot();
   279→   279→    const isWindows = process.platform === 'win32';
   280→   280→
   281→   281→    // Check if npm is available
   282→   282→    const npmCmd = isWindows ? 'npm.cmd' : 'npm';
   283→   283→
   284→   284→    try {
   285→   285→      if (await checkCloudHealth()) {
   286→   286→        cloudReady = true;
   287→   287→        cloudError = '';
   288→   288→        log('INFO', '[Launcher] Existing cloud backend detected');
   289→   289→        resolve(true);
   290→   290→        return;
   291→   291→      }
   292→   292→
   293→   293→      cloudProcess = spawn(npmCmd, ['run', 'dev', '--', '--hostname', '127.0.0.1', '--port', '3000'], {
   294→   294→        cwd: root,
   295→   295→        shell: true,
   296→   296→        stdio: ['ignore', 'pipe', 'pipe'],
   297→   297→      });
   298→   298→

   307→   307→            log(ready ? 'INFO' : 'WARN', '[Launcher]', ready ? 'Cloud backend ready' : cloudError);
   308→   308→            resolve(ready);
   309→   309→          });
   310→   310→        }
   311→   311→      }, (output) => {
   312→   312→        output = output.trim();
   313→   313→        if (output) cloudError = output.split(/\r?\n/).slice(-1)[0];
   314→   314→        log('WARN', '[Cloud]', output);
   315→   315→      });
   316→   316→
   317→   317→      cloudProcess.on('error', (err) => {
   318→   318→        cloudError = err.message;
   319→   319→        log('ERROR', '[Cloud] Process error:', err.message);
   320→   320→        reject(err);
   321→   321→      });
   322→   322→
   323→   323→      cloudProcess.on('close', (code) => {
   324→   324→        log('INFO', '[Cloud] Process exited with code:', code);
   325→   325→        cloudReady = false;

   327→   327→      });
   328→   328→
   329→   329→      // Timeout fallback
   330→   330→      setTimeout(() => {
   331→   331→        if (!cloudReady) {
   332→   332→          cloudError = cloudError || 'Next.js startup timed out on http://127.0.0.1:3000';
   333→   333→          log('WARN', '[Launcher]', cloudError);
   334→   334→          resolve(false);
   335→   335→        }
   336→   336→      }, MAX_STARTUP_WAIT);
   337→   337→
   338→   338→    } catch (err) {
   339→   339→      cloudError = err.message;
   340→   340→      log('ERROR', '[Launcher] Failed to start cloud:', err.message);
   341→   341→      reject(err);
   342→   342→    }
   343→   343→  });
   344→   344→}

   800→   800→function buildMultipart(fields, boundary) {

   802→   802→  for (const [name, value] of Object.entries(fields)) {
   803→   803→    if (value && typeof value === 'object' && value.filename) {
   804→   804→      parts.push(Buffer.from(`--${boundary}\r\nContent-Disposition: form-data; name="${name}"; filename="${value.filename}"\r\nContent-Type: ${value.contentType || 'application/octet-stream'}\r\n\r\n`, 'utf8'));
   805→   805→      parts.push(Buffer.isBuffer(value.data) ? value.data : Buffer.from(value.data));
   806→   806→      parts.push(Buffer.from('\r\n', 'utf8'));
   807→   807→    } else {
   808→   808→      parts.push(Buffer.from(`--${boundary}\r\nContent-Disposition: form-data; name="${name}"\r\n\r\n${value}\r\n`, 'utf8'));
   809→   809→    }
   810→   810→  }
   811→   811→  parts.push(Buffer.from(`--${boundary}--\r\n`, 'utf8'));
   812→   812→  return Buffer.concat(parts);
   813→   813→}

   815→   815→function httpsPostBuffer(urlStr, body, headers) {
   816→   816→  return new Promise((resolve, reject) => {
   817→   817→    const parsedUrl = new URL(urlStr);
   818→   818→    const mod = parsedUrl.protocol === 'http:' ? require('http') : require('https');
   819→   819→    const options = {
   820→   820→      hostname: parsedUrl.hostname,
   821→   821→      port: parsedUrl.port || (parsedUrl.protocol === 'http:' ? 80 : 443),
   822→   822→      path: parsedUrl.pathname + parsedUrl.search,
   823→   823→      method: 'POST',
   824→   824→      headers: { ...headers, 'Content-Length': body.length },
   825→   825→    };
   826→   826→    const req = mod.request(options, (res) => {
   827→   827→      const chunks = [];
   828→   828→      res.on('data', (chunk) => chunks.push(chunk));
   829→   829→      res.on('end', () => {
   830→   830→        const data = Buffer.concat(chunks);
   831→   831→        resolve({ statusCode: res.statusCode, headers: res.headers, data });
   832→   832→      });
   833→   833→    });
   834→   834→    req.on('error', reject);
   835→   835→    req.setTimeout(30000, () => { req.destroy(new Error('Request timeout')); });
   836→   836→    req.write(body);
   837→   837→    req.end();
   838→   838→  });
   839→   839→}
