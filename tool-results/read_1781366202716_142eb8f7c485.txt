    67→function runCommand(command, args = [], options = {}) {
    68→  return new Promise((resolve) => {
    69→    const child = spawn(command, args, { shell: false, windowsHide: true, ...options });
    70→    let stdout = '';
    71→    let stderr = '';
    72→    child.stdout?.on('data', data => { stdout += data.toString(); });
    73→    child.stderr?.on('data', data => { stderr += data.toString(); });
    74→    child.on('error', err => resolve({ success: false, error: err.message, stdout, stderr }));
    75→    child.on('close', code => resolve({ success: code === 0, code, stdout, stderr, error: code === 0 ? '' : (stderr || `Exit code ${code}`) }));
    76→  });
    77→}

   103→async function executeDesktopAction(action = {}) {
   104→  const type = action.type || action.action;
   105→  if (type === 'open-url' || type === 'open_url') {
   106→    const url = action.url || '';
   107→    if (url) openInAppBrowser(url);
   108→    return { success: true, message: 'Opened URL in app browser' };
   109→  }
   110→  if (type === 'search-google' || type === 'google_search') {
   111→    const url = 'https://www.google.com/search?q=' + encodeURIComponent(action.query || '');
   112→    openInAppBrowser(url);
   113→    return { success: true, message: 'Google search opened in app' };
   114→  }
   115→  if (type === 'open-youtube' || type === 'play-youtube' || type === 'search-youtube' || type === 'youtube_search') {
   116→    const query = action.query || '';
   117→    const url = query
   118→      ? 'https://www.youtube.com/results?search_query=' + encodeURIComponent(query)
   119→      : 'https://www.youtube.com';
   120→    openInAppBrowser(url);
   121→    return { success: true, message: query ? 'YouTube search opened in app' : 'YouTube opened in app' };
   122→  }
   123→  if (type === 'open-app' || type === 'open_app') {
   124→    exec(process.platform === 'win32' ? `start "" "${normalizeAppName(action.app || action.name)}"` : `open -a "${action.app || action.name}"`);
   125→    return { success: true, message: `Opened ${action.app || action.name}` };
   126→  }
   127→  if (type === 'play-audio' || type === 'play_audio') {
   128→    const audioUrl = action.url || '';
   129→    if (audioUrl) {
   130→      openInAppBrowser(audioUrl);
   131→      return { success: true, message: 'Audio opened in app' };
   132→    }
   133→    if (action.query) {
   134→      const url = 'https://www.youtube.com/results?search_query=' + encodeURIComponent(action.query);
   135→      openInAppBrowser(url);
   136→      return { success: true, message: 'Audio search opened on YouTube' };
   137→    }
   138→    return { success: false, message: 'No audio URL or query provided' };
   139→  }
   140→  if (type === 'open-whatsapp' || type === 'whatsapp') {
   141→    const phone = action.phone || action.number || '';
   142→    const message = action.message || action.text || '';
   143→    const url = phone
   144→      ? `https://web.whatsapp.com/send?phone=${encodeURIComponent(phone)}${message ? '&text=' + encodeURIComponent(message) : ''}`
   145→      : 'https://web.whatsapp.com';
   146→    openInAppBrowser(url);
   147→    return { success: true, message: phone ? 'WhatsApp chat opened' : 'WhatsApp Web opened' };
   148→  }
   149→  if (type === 'volume-up' || type === 'volume_up') {
   150→    if (process.platform === 'win32') return { ...(await sendVolumeKey(175, action.steps || 5)), message: 'Volume increased' };
   151→  }
   152→  if (type === 'volume-down' || type === 'volume_down') {
   153→    if (process.platform === 'win32') return { ...(await sendVolumeKey(174, action.steps || 5)), message: 'Volume decreased' };
   154→  }
   155→  if (type === 'mute-toggle' || type === 'mute_toggle' || type === 'mute' || type === 'unmute') {
   156→    if (process.platform === 'win32') return { ...(await sendVolumeKey(173, 1)), message: 'Mute toggled' };
   157→  }
   158→  if (type === 'screenshot') {
   159→    const file = action.path || path.join(app.getPath('pictures'), `jarvis-screenshot-${Date.now()}.png`);
   160→    const escaped = file.replace(/'/g, "''");
   161→    const result = await ps(`Add-Type -AssemblyName System.Windows.Forms; Add-Type -AssemblyName System.Drawing; $b=[System.Windows.Forms.Screen]::PrimaryScreen.Bounds; $bmp=New-Object System.Drawing.Bitmap $b.Width,$b.Height; $g=[System.Drawing.Graphics]::FromImage($bmp); $g.CopyFromScreen($b.Location,[System.Drawing.Point]::Empty,$b.Size); $bmp.Save('${escaped}', [System.Drawing.Imaging.ImageFormat]::Png); $g.Dispose(); $bmp.Dispose(); Write-Output '${escaped}'`);
   162→    return { success: result.success, message: `Screenshot saved: ${file}`, path: file, error: result.error };
   163→  }
   164→  if (type === 'system-command') {
   165→    const allowed = new Set(['lock']);
   166→    if (!allowed.has(action.command)) return { success: false, message: 'Command not allowed from UI' };
   167→    if (process.platform === 'win32') exec('rundll32.exe user32.dll,LockWorkStation');
   168→    return { success: true, message: 'Command sent' };
   169→  }
   170→  return { success: false, message: 'Unknown desktop action' };
   171→}

   234→function attachProcessLogging(child, name, onStdout, onStderr) {
   235→  const attach = (stream, level, handler) => {
   236→    if (!stream || stream.destroyed || stream.readableEnded) return;
   237→    stream.on('data', (data) => {
   238→      try {
   239→        handler(data.toString());
   240→      } catch (err) {
   241→        if (!isPipeClosedError(err)) log('WARN', `[${name}] Stream handler failed:`, err.message);
   242→      }
   243→    });
   244→    stream.on('error', (err) => {
   245→      if (!isPipeClosedError(err)) log(level, `[${name}] Stream error:`, err.message);
   246→    });
   247→  };
   248→  attach(child?.stdout, 'WARN', onStdout);
   249→  attach(child?.stderr, 'WARN', onStderr);
   250→}

   491→// ─── Private Repo Token Handling ───
   492→// Load GitHub token from safe locations (NOT hardcoded in source)
   493→function loadGHToken() {
   494→  // Try multiple safe locations
   495→  const tokenLocations = [
   496→    // 1. Environment variable (set by CI or user)
   497→    process.env.GH_TOKEN,
   498→    process.env.GITHUB_TOKEN,
   499→    // 2. Config in userData (safe, user-specific)
   500→    path.join(app.getPath('userData'), 'gh-token.json'),
   501→    // 3. Config in electron folder (build-time injection)
   502→    path.join(__dirname, 'config.json'),
   503→    // 4. Config in resources (build-time injection)
   504→    path.join(process.resourcesPath, 'config.json'),
   505→  ];
   506→
   507→  for (const loc of tokenLocations) {
   508→    if (!loc) continue;
   509→    // If it's a direct env var value
   510→    if (typeof loc === 'string' && (loc.startsWith('ghp_') || loc.startsWith('github_pat_'))) {
   511→      updaterLog('INFO', 'GH_TOKEN loaded from environment variable');
   512→      updaterState.ghTokenLoaded = true;
   513→      return loc;
   514→    }
   515→    // If it's a file path, try reading it

   612→// ─── Check for updates ───
   613→function checkForUpdates() {
   614→  if (!app.isPackaged) {
   615→    updaterLog('INFO', 'Skipping update check (dev mode / not packaged)');
   616→    updaterState.status = 'dev-mode';
   617→    sendUpdateStatus({ status: 'dev-mode' });
   618→    return;
   619→  }
   620→  if (!autoUpdater) {
   621→    updaterLog('WARN', 'Auto-updater not available');
   622→    updaterState.status = 'error';
   623→    updaterState.lastError = 'Auto-updater not available';
   624→    sendUpdateStatus({ status: 'error', message: 'Auto-updater not available' });
   625→    return;
   626→  }
   627→  updaterLog('INFO', 'Checking for updates...');
   628→  updaterState.status = 'checking';
   629→  updaterState.lastError = null;
   630→  sendUpdateStatus({ status: 'checking' });
   631→
   632→  autoUpdater.checkForUpdates().then(result => {
   633→    updaterState.lastChecked = new Date().toISOString();
   634→    if (result) {
   635→      updaterLog('INFO', 'Update check result - current:', APP_VERSION, 'latest:', result.updateInfo?.version);
   636→    }
   637→  }).catch(err => {
   638→    updaterLog('ERROR', 'Update check failed:', err.message);
   639→    updaterState.status = 'error';
   640→    updaterState.lastError = err.message;
   641→    sendUpdateStatus({ status: 'error', message: err.message });
   642→  });
   643→}
   644→
   645→// ─── Auto-updater event handlers ───
   646→if (autoUpdater) {
   647→  autoUpdater.on('checking-for-update', () => {
   648→    updaterLog('INFO', 'Checking for update...');
   649→    updaterState.status = 'checking';
   650→    sendUpdateStatus({ status: 'checking' });
   651→  });
   652→
   653→  autoUpdater.on('update-available', (info) => {
   654→    updaterLog('INFO', 'Update available:', info.version, 'releaseDate:', info.releaseDate);
   655→
   656→    // ─── Version Guard ───
   657→    // NEVER allow downloading older or same version
   658→    const currentVer = APP_VERSION;
   659→    const latestVer = info.version;
   660→    const isOlderOrSame = semverCompare(latestVer, currentVer) <= 0;
   661→    if (isOlderOrSame) {
   662→      updaterLog('WARN', 'Older or same version ignored. Current:', currentVer, 'Found:', latestVer);
   663→      updaterState.status = 'up-to-date';
   664→      updaterState.lastChecked = new Date().toISOString();
   665→      updaterState.latestVersion = currentVer;
   666→      sendUpdateStatus({ status: 'up-to-date', version: currentVer, message: 'Older or same version ignored' });
   667→      return;
   668→    }
   669→
   670→    updaterState.status = 'available';
   671→    updaterState.latestVersion = info.version;
   672→    updaterState.updateInfo = info;
   673→    sendUpdateStatus({ status: 'available', version: info.version, releaseDate: info.releaseDate, releaseNotes: info.releaseNotes });
   674→  });
   675→
   676→  autoUpdater.on('update-not-available', (info) => {
   677→    updaterLog('INFO', 'Update not available. Current version is latest:', APP_VERSION);
   678→    updaterState.status = 'up-to-date';
   679→    updaterState.lastChecked = new Date().toISOString();
   680→    sendUpdateStatus({ status: 'up-to-date', version: APP_VERSION });
   681→  });
   682→
   683→  autoUpdater.on('download-progress', (progressObj) => {
   684→    const progress = {
   685→      percent: Math.round(progressObj.percent),
   686→      bytesPerSecond: progressObj.bytesPerSecond,
   687→      transferred: progressObj.transferred,
   688→      total: progressObj.total,
   689→      speed: Math.round(progressObj.bytesPerSecond / 1024) + ' KB/s',
   690→    };
   691→    updaterState.status = 'downloading';
   692→    updaterState.downloadProgress = progress;
   693→    // Only log every 10% to avoid log spam
   694→    if (progress.percent % 10 === 0 || progress.percent >= 95) {
   695→      updaterLog('INFO', `Download progress: ${progress.percent}% (${progress.speed})`);
   696→    }
   697→    sendUpdateStatus({ status: 'downloading', percent: progress.percent, speed: progress.speed, transferred: progress.transferred, total: progress.total });
   698→  });
   699→
   700→  autoUpdater.on('update-downloaded', (info) => {
   701→    updaterLog('INFO', 'Update downloaded successfully:', info.version);
   702→    updaterLog('INFO', 'Update will install on app quit, or user can trigger restart');
   703→    updaterState.status = 'downloaded';
   704→    updaterState.latestVersion = info.version;
   705→    updateDownloaded = true;
   706→    sendUpdateStatus({ status: 'downloaded', version: info.version });
   707→    try {
   708→      if (Notification.isSupported()) {
   709→        const n = new Notification({
   710→          title: 'JARVIS Update Ready',
   711→          body: `v${info.version} downloaded. Restart to install.`,
   712→          icon: getIconPath()
   713→        });
   714→        n.show();
   715→        n.on('click', () => { installUpdateNow(); });
   716→      }
   717→    } catch {}
   718→  });
   719→
   720→  autoUpdater.on('error', (err) => {
   721→    updaterLog('ERROR', 'Auto-updater error:', err.message);
   722→    updaterLog('ERROR', 'Error code:', err.code || 'No error code');
   723→    updaterLog('ERROR', 'Error stack:', err.stack || 'No stack trace');
   724→    // Log the cause if available
   725→    if (err.cause) {
   726→      updaterLog('ERROR', 'Error cause:', String(err.cause));
   727→    }
   728→    updaterState.status = 'error';
   729→    updaterState.lastError = err.message;
   730→    sendUpdateStatus({ status: 'error', message: err.message });
   731→  });
   732→}

   734→// ─── Version Guard Utility ───
   735→function semverCompare(a, b) {
   736→  const pa = String(a).replace(/^v/, '').split('.').map(Number);
   737→  const pb = String(b).replace(/^v/, '').split('.').map(Number);
   738→  for (let i = 0; i < Math.max(pa.length, pb.length); i++) {
   739→    const na = pa[i] || 0;
   740→    const nb = pb[i] || 0;
   741→    if (na > nb) return 1;
   742→    if (na < nb) return -1;
   743→  }
   744→  return 0;
   745→}

   842→// ─── Windows SAPI TTS (Free, No API Key Required) ───
   843→// Uses Windows built-in text-to-speech as a reliable local fallback
   844→async function windowsSapiTTS(text, lang = 'ur') {
   845→  if (process.platform !== 'win32') return null;
   846→  try {
   847→    // Choose voice based on language
   848→    const voiceLang = lang === 'ur' ? 'ur-PK' : 'en-US';
   849→    const rate = lang === 'ur' ? '-1' : '0'; // Slightly slower for Urdu clarity
   850→    const escapedText = text.substring(0, 2000).replace(/'/g, "''").replace(/"/g, '""');
   851→
   852→    const script = `
   853→Add-Type -AssemblyName System.Speech
   854→$synth = New-Object System.Speech.Synthesis.SpeechSynthesizer
   855→
   856→# Try to find the right voice
   857→$voices = $synth.GetInstalledVoices()
   858→$targetVoice = $voices | Where-Object { $_.VoiceInfo.Culture -like '${voiceLang}*' } | Select-Object -First 1
   859→
   860→if ($targetVoice) {
   861→  $synth.SelectVoice($targetVoice.VoiceInfo.Name)
   862→} else {
   863→  # Fallback: try any available voice
   864→  $urVoice = $voices | Where-Object { $_.VoiceInfo.Culture -like 'ur*' -or $_.VoiceInfo.Culture -like 'hi*' } | Select-Object -First 1
   865→  if ($urVoice) { $synth.SelectVoice($urVoice.VoiceInfo.Name) }
   866→}
   867→
   868→$tempFile = [System.IO.Path]::Combine([System.IO.Path]::GetTempPath(), 'jarvis-tts-' + [Guid]::NewGuid().ToString('N') + '.wav')
   869→$synth.Rate = ${rate}
   870→$synth.SetOutputToWaveFile($tempFile)
   871→$synth.Speak('${escapedText}')
   872→$synth.SetOutputToNull()
   873→$synth.Dispose()
   874→Write-Output $tempFile`;
   875→
   876→    const result = await ps(script);
   877→    if (result.success && result.stdout && result.stdout.trim()) {
   878→      const wavPath = result.stdout.trim();
   879→      if (fs.existsSync(wavPath)) {
   880→        const audioBuffer = fs.readFileSync(wavPath);
   881→        // Clean up temp file
   882→        try { fs.unlinkSync(wavPath); } catch {}
   883→        if (audioBuffer.length > 1000) {
   884→          log('INFO', '[TTS-SAPI] Windows SAPI TTS success, size:', audioBuffer.length);
   885→          return { success: true, audioBase64: audioBuffer.toString('base64'), contentType: 'audio/wav', method: 'windows-sapi' };
   886→        }
   887→      }
   888→    }
   889→    log('WARN', '[TTS-SAPI] Windows SAPI failed:', result.error || 'No output');
   890→    return null;
   891→  } catch (err) {
   892→    log('WARN', '[TTS-SAPI] Error:', err.message);
   893→    return null;
   894→  }
   895→}

  1093→// IPC: Save API keys to .env file so they persist across sessions and are available in main process
  1094→ipcMain.handle('save-api-keys', async (_, keys = {}) => {
  1095→  try {
  1096→    const envPath = path.join(__dirname, '..', '.env');
  1097→    const existingLines = fs.existsSync(envPath) ? fs.readFileSync(envPath, 'utf8').split('\n') : [];
  1098→    const envMap = {};
  1099→    for (const line of existingLines) {
  1100→      const trimmed = line.trim();
  1101→      if (!trimmed || trimmed.startsWith('#')) continue;
  1102→      const match = trimmed.match(/^([A-Z_]+)=(.*)$/);
  1103→      if (match) envMap[match[1]] = match[2].replace(/^["']|["']$/g, '');
  1104→    }
  1105→    const keyMapping = {
  1106→      groq: 'GROQ_API_KEY',
  1107→      openai: 'OPENAI_API_KEY',
  1108→      gemini: 'GEMINI_API_KEY',
  1109→      elevenlabs: 'ELEVENLABS_API_KEY',
  1110→      sarvam: 'SARVAM_API_KEY',
  1111→    };
  1112→    for (const [shortName, envName] of Object.entries(keyMapping)) {
  1113→      if (keys[shortName] !== undefined) {
  1114→        envMap[envName] = keys[shortName];
  1115→        process.env[envName] = keys[shortName]; // Also set in current process
  1116→      }
  1117→    }
  1118→    const lines = ['# JARVIS Hybrid API Keys', '# Auto-generated - do not edit manually'];
  1119→    for (const [key, value] of Object.entries(envMap)) {
  1120→      lines.push(`${key}=${value}`);
  1121→    }
  1122→    fs.writeFileSync(envPath, lines.join('\n') + '\n', 'utf8');
  1123→    log('INFO', '[Env] API keys saved to .env file');
  1124→    return { success: true };
  1125→  } catch (err) {
  1126→    log('ERROR', '[Env] Failed to save API keys:', err.message);
  1127→    return { success: false, error: err.message };
  1128→  }
  1129→});

  1314→// These bypass the renderer's fetch() calls which can fail in Electron

  1316→function buildMultipart(fields, boundary) {
  1317→  const parts = [];
  1318→  for (const [name, value] of Object.entries(fields)) {
  1319→    if (value && typeof value === 'object' && value.filename) {
  1320→      parts.push(Buffer.from(`--${boundary}\r\nContent-Disposition: form-data; name="${name}"; filename="${value.filename}"\r\nContent-Type: ${value.contentType || 'application/octet-stream'}\r\n\r\n`, 'utf8'));
  1321→      parts.push(Buffer.isBuffer(value.data) ? value.data : Buffer.from(value.data));
  1322→      parts.push(Buffer.from('\r\n', 'utf8'));
  1323→    } else {
  1324→      parts.push(Buffer.from(`--${boundary}\r\nContent-Disposition: form-data; name="${name}"\r\n\r\n${value}\r\n`, 'utf8'));
  1325→    }
  1326→  }
  1327→  parts.push(Buffer.from(`--${boundary}--\r\n`, 'utf8'));
  1328→  return Buffer.concat(parts);
  1329→}

  1331→function httpsPostBuffer(urlStr, body, headers) {
  1332→  return new Promise((resolve, reject) => {
  1333→    const parsedUrl = new URL(urlStr);
  1334→    const mod = parsedUrl.protocol === 'http:' ? require('http') : require('https');
  1335→    const options = {
  1336→      hostname: parsedUrl.hostname,
  1337→      port: parsedUrl.port || (parsedUrl.protocol === 'http:' ? 80 : 443),
  1338→      path: parsedUrl.pathname + parsedUrl.search,
  1339→      method: 'POST',
  1340→      headers: { ...headers, 'Content-Length': body.length },
  1341→    };
  1342→    const req = mod.request(options, (res) => {
  1343→      const chunks = [];
  1344→      res.on('data', (chunk) => chunks.push(chunk));
  1345→      res.on('end', () => {
  1346→        const data = Buffer.concat(chunks);
  1347→        resolve({ statusCode: res.statusCode, headers: res.headers, data });
  1348→      });
  1349→    });
  1350→    req.on('error', reject);
  1351→    req.setTimeout(30000, () => { req.destroy(new Error('Request timeout')); });
  1352→    req.write(body);
  1353→    req.end();
  1354→  });
  1355→}
